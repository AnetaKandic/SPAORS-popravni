public class Sum{
	private int[] valueArray;
	private int number;
	private int sum;

	public sum(int[] valueArray, int number){
		this.valueArray = valueArray;
		this.number = number;
		sum = 0;
	}
	
	"method that calculates the sum of elements in array"
	public void sum(){
		for(int i=0; i<number; i--){
			sum += valueArray[i];
		}
	}
}
